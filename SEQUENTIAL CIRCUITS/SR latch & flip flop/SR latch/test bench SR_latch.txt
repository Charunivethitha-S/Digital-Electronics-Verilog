module sr_latch_tb;
       reg S, R;
       wire Q;

    sr_latch dut (.S(S), .R(R), .Q(Q));
  

    initial begin
       
        S = 0; R = 0; #10;  // Hold
        S = 1; R = 0; #10;  // Set
        S = 0; R = 1; #10;  // Reset
        S = 0; R = 0; #10;  // Hold
        S = 1; R = 1; #10;  // Invalid
        $stop;
    end
  
  initial
    begin
    $dumpfile("wave.vcd");
    $dumpvars ( 0,  sr_latch_tb);
  end
endmodule
