module test_bench;
  reg clk, j, k;
  wire q, qbar;
  
  jk_flip_flop JK ( clk, j, k, q, qbar);
  
  always #5 clk = ~clk;
  module jk_flip_flop ( input clk, j, k, output reg q, qbar);
  
  assign qbar = ~q;
  
  always@(*)
    begin
      case ( {j,k} )
        2'b00 : q <= q;
        2'b01 : q <= 1'b0;
        2'b10 : q <= 1'b1;
        2'b11 : q <= ~q;
        default : q <= 1'bx;
      endcase
    end
endmodule
  initial
    begin
  {clk,j,k} = 0;
  end
  
  initial
    begin
  j = 0; k = 0; #10;
  j = 0; k = 1; #10;
  j = 1; k = 0; #10;
  j = 1; k = 1; #10;
      $finish;
    end
  
  initial
    begin
      $dumpfile ( "wave.vcd" );
      $dumpvars ( 0, test_bench);
    end
  
endmodule
