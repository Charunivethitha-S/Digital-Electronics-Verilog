`timescale 1ns/1ps

module piso_shift_register_tb;

reg clk;
reg reset;
reg load;
reg [3:0] p_in;
wire s_out;

piso_shift_register uut (
    .clk(clk),
    .reset(reset),
    .load(load),
    .p_in(p_in),
    .s_out(s_out)
);


always #5 clk = ~clk;

initial begin
    
    clk   = 0;
    reset = 1;
    load  = 0;
    p_in  = 4'b0000;

    #10 reset = 0;

   
    #10 load = 1;
        p_in = 4'b1011;

    #10 load = 0;  

    #40;            

  
    #10 load = 1;
        p_in = 4'b1101;

    #10 load = 0;

    #40 $finish;
end
  
  initial
    begin
      $dumpfile ("wave.vcd");
                 $dumpvars (0,piso_shift_register_tb);
end

endmodule
