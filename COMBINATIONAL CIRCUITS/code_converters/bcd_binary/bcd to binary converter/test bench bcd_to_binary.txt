module tb_bcd_to_binary;

    reg [3:0] bcd;
    wire [3:0] bin;

    // Instantiate DUT
    bcd_to_binary uut (
        .bcd(bcd),
        .bin(bin)
    );

    initial begin
        // Apply all valid BCD inputs
        bcd = 4'b0000; #10;
        bcd = 4'b0001; #10;
        bcd = 4'b0010; #10;
        bcd = 4'b0011; #10;
        bcd = 4'b0100; #10;
        bcd = 4'b0101; #10;
        bcd = 4'b0110; #10;
        bcd = 4'b0111; #10;
        bcd = 4'b1000; #10;
        bcd = 4'b1001; #10;

        // Invalid BCD test
        bcd = 4'b1010; #10;

        $stop;
    end

  initial
    begin
      $dumpfile ("wave.vcd");
      $dumpvars (0, tb_bcd_to_binary);
    end
     
endmodule
