module tb_binary_to_bcd;
    reg [7:0] binary;
    wire [11:0] bcd;

    binary_to_bcd uut (.binary(binary), .bcd(bcd));

    initial begin
        binary = 8'd0;   #10;
        binary = 8'd7;   #10;
        binary = 8'd25;  #10;
        binary = 8'd99;  #10;
        binary = 8'd123; #10;
        binary = 8'd255; #10;
        $stop;
    end
  
  initial
    begin
      $dumpfile ("wave.vcd");
      $dumpvars(0, tb_binary_to_bcd);
    end
endmodule
