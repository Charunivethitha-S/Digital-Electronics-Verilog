module tb_excess3_to_bcd;

    reg [3:0] excess3;
    wire [3:0] bcd;

    // Instantiate DUT
    excess3_to_bcd uut (
        .excess3(excess3),
        .bcd(bcd)
    );

    initial begin
        excess3 = 4'b0011; #10; 
        excess3 = 4'b0100; #10; 
        excess3 = 4'b0101; #10;
        excess3 = 4'b0110; #10; 
        excess3 = 4'b0111; #10; 
        excess3 = 4'b1000; #10; 
        excess3 = 4'b1001; #10; 
        excess3 = 4'b1010; #10;
        excess3 = 4'b1011; #10; 
        excess3 = 4'b1100; #10; 
        excess3 = 4'b1110; #10; 

        $stop;
      
    end
  
  initial
    begin
      $dumpfile ("wave.vcd");
      $dumpvars (0,  tb_excess3_to_bcd);
    end
  
endmodule
